def flag_calibrators(df: pd.DataFrame) -> pd.DataFrame:
    out = df.copy()
    if "is_calibrator" not in out.columns:
        out["is_calibrator"] = False
    out["is_calibrator"] = out["is_calibrator"].fillna(False).astype(bool)
    return out

def separate_calibrators(df: pd.DataFrame):
    out = flag_calibrators(df)
    non_calibrators = out.loc[~out["is_calibrator"]].copy().reset_index(drop=True)
    calibrators = out.loc[out["is_calibrator"]].copy().reset_index(drop=True)
    return non_calibrators, calibrators

3) new calibration.py

from __future__ import annotations
import pandas as pd

def summarize_calibrators(calibrator_df: pd.DataFrame, *, plate_col="plate_id", target_col="target", ct_col="ct", calibrator_id_col="calibrator_id") -> pd.DataFrame:
    required = [plate_col, target_col, ct_col]
    missing = [col for col in required if col not in calibrator_df.columns]
    if missing:
        raise ValueError(f"Missing required calibrator columns: {missing}")

    out = calibrator_df.copy()
    if "is_calibrator" in out.columns:
        out = out.loc[out["is_calibrator"].fillna(False).astype(bool)].copy()
    out = out.loc[out[ct_col].notna()].copy()

    group_cols = [plate_col, target_col]
    if calibrator_id_col in out.columns and out[calibrator_id_col].notna().any():
        group_cols.append(calibrator_id_col)

    return (
        out.groupby(group_cols, dropna=False)[ct_col]
        .agg(["mean", "std", "count"])
        .reset_index()
        .rename(columns={"mean": "calibrator_ct_mean", "std": "calibrator_ct_std", "count": "n_calibrator_reps"})
    )

def calculate_plate_calibrator_offsets(calibrator_summary_df: pd.DataFrame, *, plate_col="plate_id", target_col="target", calibrator_mean_col="calibrator_ct_mean", reference_method="global_mean", reference_plate=None) -> pd.DataFrame:
    required = [plate_col, target_col, calibrator_mean_col]
    missing = [col for col in required if col not in calibrator_summary_df.columns]
    if missing:
        raise ValueError(f"Missing required columns in calibrator summary: {missing}")

    out = calibrator_summary_df.copy()

    if reference_method == "global_mean":
        ref = (
            out.groupby(target_col, dropna=False)[calibrator_mean_col]
            .mean().reset_index(name="reference_calibrator_ct")
        )
    elif reference_method == "reference_plate":
        if reference_plate is None:
            raise ValueError("reference_plate must be provided when reference_method='reference_plate'.")
        ref = (
            out.loc[out[plate_col] == reference_plate, [target_col, calibrator_mean_col]]
            .rename(columns={calibrator_mean_col: "reference_calibrator_ct"})
            .drop_duplicates(subset=[target_col])
        )
    else:
        raise ValueError("reference_method must be 'global_mean' or 'reference_plate'.")

    out = out.merge(ref, on=target_col, how="left", validate="many_to_one")
    out["offset_ct"] = out[calibrator_mean_col] - out["reference_calibrator_ct"]
    return out

def apply_plate_calibration(df: pd.DataFrame, offsets_df: pd.DataFrame, *, plate_col="plate_id", target_col="target", ct_col="ct", offset_col="offset_ct", output_col="ct_calibrated", keep_calibrator_rows=True) -> pd.DataFrame:
    out = df.copy()
    out = out.merge(
        offsets_df[[plate_col, target_col, offset_col]].drop_duplicates(),
        on=[plate_col, target_col],
        how="left",
        validate="many_to_one",
    )
    out[output_col] = out[ct_col]
    mask = out[ct_col].notna() & out[offset_col].notna()
    out.loc[mask, output_col] = out.loc[mask, ct_col] - out.loc[mask, offset_col]

    if not keep_calibrator_rows and "is_calibrator" in out.columns:
        out = out.loc[~out["is_calibrator"].fillna(False).astype(bool)].copy()

    return out

from qpcr.preprocess import separate_calibrators
from qpcr.calibration import (
    summarize_calibrators,
    calculate_plate_calibrator_offsets,
    apply_plate_calibration,
)

non_cal_df, calibrator_df = separate_calibrators(df)

cal_summary_df = summarize_calibrators(calibrator_df)
cal_offsets_df = calculate_plate_calibrator_offsets(
    cal_summary_df,
    reference_method="global_mean",
)

df = apply_plate_calibration(df, cal_offsets_df, output_col="ct_calibrated")
